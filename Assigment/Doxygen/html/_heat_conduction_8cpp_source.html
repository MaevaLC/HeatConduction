212 void ImplicitMethod::ThomasAlgorith() {
213  //Forward elimination phase
214  for (int k = 1; k < s - 1; k++){
215  m = a[k] / b[k - 1];
216  b[k] = b[k] - (m*c[k - 1]);
217  d[k] = d[k] - (m*d[k - 1]);
218  }
219 
220  //Backward elimination phase
221  u_nplus1[s] = Text_0;
222  u_nplus1[0] = Text_0;
223  u_nplus1[s - 1] = d[s - 2] / b[s - 2];
224  for (int k = s - 3; k > -1; k--){
225  u_nplus1[k + 1] = (d[k] - (c[k] * u_nplus1[k + 2])) / b[k];
226  }

229 }
